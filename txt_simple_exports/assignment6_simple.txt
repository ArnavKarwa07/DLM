import numpy as np
import tensorflow as tf


def prepare_data(size=3000, img_size=96):
    (x_train, y_train), (x_test, y_test) = tf.keras.datasets.cifar10.load_data()
    x_train = x_train[:size]
    y_train = y_train[:size].ravel()
    x_test = x_test[:1000]
    y_test = y_test[:1000].ravel()

    x_train = tf.image.resize(x_train, (img_size, img_size)).numpy()
    x_test = tf.image.resize(x_test, (img_size, img_size)).numpy()

    x_train = tf.keras.applications.mobilenet_v2.preprocess_input(x_train)
    x_test = tf.keras.applications.mobilenet_v2.preprocess_input(x_test)
    return x_train, y_train, x_test, y_test


def build_model(img_size=96, num_classes=10):
    try:
        base = tf.keras.applications.MobileNetV2(
            input_shape=(img_size, img_size, 3),
            include_top=False,
            weights="imagenet",
        )
        print("Using pretrained ImageNet weights.")
    except Exception as e:
        print("Could not load pretrained weights, fallback to random init:", e)
        base = tf.keras.applications.MobileNetV2(
            input_shape=(img_size, img_size, 3),
            include_top=False,
            weights=None,
        )

    base.trainable = False
    model = tf.keras.Sequential([
        tf.keras.layers.Input(shape=(img_size, img_size, 3)),
        base,
        tf.keras.layers.GlobalAveragePooling2D(),
        tf.keras.layers.Dense(128, activation="relu"),
        tf.keras.layers.Dropout(0.2),
        tf.keras.layers.Dense(num_classes, activation="softmax"),
    ])
    return model, base


def main():
    tf.random.set_seed(42)

    x_train, y_train, x_test, y_test = prepare_data()
    model, base = build_model()

    model.compile(optimizer=tf.keras.optimizers.Adam(1e-3), loss="sparse_categorical_crossentropy", metrics=["accuracy"])
    model.fit(x_train, y_train, epochs=2, batch_size=64, validation_split=0.1, verbose=0)

    # Light fine-tuning.
    base.trainable = True
    for layer in base.layers[:-20]:
        layer.trainable = False

    model.compile(optimizer=tf.keras.optimizers.Adam(5e-5), loss="sparse_categorical_crossentropy", metrics=["accuracy"])
    model.fit(x_train, y_train, epochs=1, batch_size=64, validation_split=0.1, verbose=0)

    loss, acc = model.evaluate(x_test, y_test, verbose=0)
    print("Test loss:", round(float(loss), 4))
    print("Test accuracy:", round(float(acc), 4))

    preds = np.argmax(model.predict(x_test[:10], verbose=0), axis=1)
    print("Sample predictions:", preds.tolist())
    print("Sample labels:", y_test[:10].tolist())


if __name__ == "__main__":
    main()
