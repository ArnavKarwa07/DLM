import tensorflow as tf
from pathlib import Path

node1=tf.constant(3.0)
node2=tf.constant(4.0)
print(node1,node2)

with tf.compat.v1.Session() as sess:
    a=tf.constant(3,name="input_a")
    b=tf.constant(4,name="input_b")
    c=tf.add(a,b,name="Result")
    writer=tf.compat.v1.summary.FileWriter(str(Path("./logs")),sess.graph)
    print(sess.run(c))
    sess.close()
    writer.close()


print("TensorBoard logs written to ./logs")
